def environment_capabilities
  browser = ENV['SAUCE_BROWSER']
  version = ENV['SAUCE_VERSION']
  platform = ENV['SAUCE_PLATFORM']

  if browser && version && platform
    return {
        :browserName => browser,
        :version => version,
        :platform => platform
    }
  end

  return nil
end
